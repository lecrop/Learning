matt retr_rnaseq <TABLE> [-fasta] [-keepsra] [-o <OUTDIR>] [-p <N_PROCS>]

   ... retrieves RNAseq SRA archives from GEO, and extracts reads from them.

   <TABLE>      : tab-/blank-/comma-/semicolon-separated table without header.
                    * first column:  GEO accession number of SRA archive containing RNAseq data, e.g., SRR3532927
                    * second column: (if available) stem of file names for downloaded data, e.g., KD1, WT1 (no blanks in stems!)
                      If second column ommited, files will be named after accession number.
   -fasta       : by default, FASTQ files will be extracted. If you give -fasta, then FASTA files will be extracted.
   -keepsra     : by default, downloaded SRA archives will be deleted. If you give -keepsra, they will be kept.
   -o <OUTDIR>  : by default, all files will be put in current working directory. If you give -o OUTDIR, all files will
                    be stored in folder OUTDIR.
   -p <N_PROCS> : by default, one process will be started and all SRA archives will be downloaded sequentially. If you
                    specify -p N_PROCS, N_PROCS parallel processes will download achrives and extract reads.

   Output: all files into current working directory or OUTDIR if speficied, and a table with information on all
             downloaded RNAseq data sets into dataset_info.tab in OUTDIR. If, by chance, files with identical file names
             already exist in OUTDIR, a file with random file name will be saved, and you need to solve the naming issue later.

function (package = NULL)
{
    z <- list()
    z$R.version <- R.Version()
    z$platform <- z$R.version$platform
    if (nzchar(.Platform$r_arch))
        z$platform <- paste(z$platform, .Platform$r_arch, sep = "/")
    z$platform <- paste0(z$platform, " (", 8 * .Machine$sizeof.pointer,
        "-bit)")
    z$locale <- Sys.getlocale()
    if (.Platform$OS.type == "windows") {
        z$running <- win.version()
    }
    else if (nzchar(Sys.which("uname"))) {
        uname <- system("uname -a", intern = TRUE)
        os <- sub(" .*", "", uname)
        z$running <- switch(os, Linux = if (file.exists("/etc/os-release")) {
            tmp <- readLines("/etc/os-release")
            t2 <- if (any(startsWith(tmp, "PRETTY_NAME="))) sub("^PRETTY_NAME=",
                "", grep("^PRETTY_NAME=", tmp, value = TRUE)[1L]) else if (any(startsWith(tmp,
                "NAME"))) sub("^NAME=", "", grep("^NAME=", tmp,
                value = TRUE)[1L]) else "Linux (unknown distro)"
            sub("\"(.*)\"", "\\1", t2)
        } else if (file.exists("/etc/system-release")) {
            readLines("/etc/system-release")
        }, Darwin = {
            ver <- readLines("/System/Library/CoreServices/SystemVersion.plist")
            ind <- grep("ProductUserVisibleVersion", ver)
            ver <- ver[ind + 1L]
            ver <- sub(".*<string>", "", ver)
            ver <- sub("</string>$", "", ver)
            ver1 <- strsplit(ver, ".", fixed = TRUE)[[1L]][2L]
            sprintf("%s %s %s", ifelse(as.numeric(ver1) < 12,
                "OS X", "macOS"), switch(ver1, `6` = "Snow Leopard",
                `7` = "Lion", `8` = "Mountain Lion", `9` = "Mavericks",
                `10` = "Yosemite", `11` = "El Capitan", `12` = "Sierra",
                `13` = "High Sierra", ""), ver)
        }, SunOS = {
            ver <- system("uname -r", intern = TRUE)
            paste("Solaris", strsplit(ver, ".", fixed = TRUE)[[1L]][2L])
        }, uname)
    }
    if (is.null(package)) {
        package <- grep("^package:", search(), value = TRUE)
        keep <- vapply(package, function(x) x == "package:base" ||
            !is.null(attr(as.environment(x), "path")), NA)
        package <- .rmpkg(package[keep])
    }
    pkgDesc <- lapply(package, packageDescription, encoding = NA)
    if (length(package) == 0)
        stop("no valid packages were specified")
    basePkgs <- sapply(pkgDesc, function(x) !is.null(x$Priority) &&
        x$Priority == "base")
    z$basePkgs <- package[basePkgs]
    if (any(!basePkgs)) {
        z$otherPkgs <- pkgDesc[!basePkgs]
        names(z$otherPkgs) <- package[!basePkgs]
    }
    loadedOnly <- loadedNamespaces()
    loadedOnly <- loadedOnly[!(loadedOnly %in% package)]
    if (length(loadedOnly)) {
        names(loadedOnly) <- loadedOnly
        pkgDesc <- c(pkgDesc, lapply(loadedOnly, packageDescription))
        z$loadedOnly <- pkgDesc[loadedOnly]
    }
    z$matprod <- as.character(options("matprod"))
    es <- extSoftVersion()
    z$BLAS <- as.character(es["BLAS"])
    z$LAPACK <- La_library()
    class(z) <- "sessionInfo"
    z
}
